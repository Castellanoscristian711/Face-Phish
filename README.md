# Instalacion en Linux

$ sudo apt-get upgrade -y

$ sudo apt-get install php git 

$ sudo apt-get install wget

$ cd $HOME

$ git clone https://github.com/edgarluck/Face-Phish.git

$ cd Face-Phish

$ bash face-phish.sh

# Instalacion en Termux
